======================
Liquid directive
======================

{% liquid
  assign names = 'Bob, Sally' | split: ', '

  for name in names
    echo 'Hello, ' | append: name
    unless forloop.last
      echo ', '
    endunless
  endfor
%}
---

(template
  (content)
  (liquid_directive
    (liquid_assign_statement
      left: (identifier)
      right: (filtered_value
        operand: (string)
        filter: (filter_expression
          name: (identifier)
          parameter: (string)
        )
      )
    )

    (liquid_for_block
      (liquid_for_statement
        left: (identifier)
        right: (identifier)
      )
      (liquid_echo_statement
        (filtered_value
          operand: (string)
          filter: (filter_expression
            name: (identifier)
            parameter: (identifier)
          )
        )
      )
      (liquid_unless_block
        (liquid_unless_statement
          (selector_expression
            operand: (identifier)
            field: (identifier)
          )
        )
        (liquid_echo_statement (string))
        (liquid_endunless_statement)
      )
      (liquid_endfor_statement)
    )
  )
)
