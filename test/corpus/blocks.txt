======================
If block
======================

{% if foo == "FOO" %}
    Variable `foo` equals "FOO"
{% else %}
    Variable `foo` not equals "FOO"
{% endif %}

---

======================
Case block
======================

{% case handle %}
  {% when "cake" %}
     This is a cake
  {% when "cookie", "biscuit" %}
     This is a cookie
  {% else %}
     This is not a cake nor a cookie
{% endcase %}

---

(template
  (content)
  (case_block
    (case_directive (identifier))
    (when_directive (string))
    (content)
    (when_directive (string) (string))
    (content)
    (else_directive)
    (content)
    (endcase_directive)
  )
  (content)
)

---
